/* ════ UPLOAD ════ */
async function uploadLecture(){
  const title=document.getElementById('lecTitle').value.trim();
  const subj=document.getElementById('lecSubject').value;
  const link=document.getElementById('lecLink').value.trim();
  const date=document.getElementById('lecDate').value;
  const img=document.getElementById('lecImage').value.trim();
  const dbp=document.getElementById('lecDb').value;
  const btn=document.getElementById('uploadBtn');
  if(!title){toast('Enter a title','err');return}
  if(!subj){toast('Select a subject','err');return}
  if(!link){toast('Enter a YouTube link','err');return}
  if(!link.includes('youtube.com')&&!link.includes('youtu.be')){toast('Invalid YouTube URL','err');return}
  btn.disabled=true;btn.innerHTML='<span class="spin"></span> Uploading...';
  const payload={title,subject:subj,youtubeLink:link,customDate:date||new Date().toISOString().split('T')[0],addedAt:new Date().toISOString()};
  if(img)payload.lectureImage=img; /* Only store if provided */
  let ok=false,used='';
  if(dbp==='db2'){try{await db2.ref('lectures').push({...payload,source:'db2'});ok=true;used='Secondary DB'}catch(e){toast('Secondary DB failed: '+e.code,'err')}}
  else if(dbp==='db1'){try{await db1.ref('lectures').push({...payload,source:'db1'});ok=true;used='Primary DB'}catch(e){toast('Primary DB failed: '+e.code,'err')}}
  else{try{await db1.ref('lectures').push({...payload,source:'db1'});ok=true;used='Primary DB'}catch(e1){toast('Primary failed, trying fallback...','info');try{await db2.ref('lectures').push({...payload,source:'db2'});ok=true;used='Secondary DB (fallback)'}catch(e2){toast('Both DBs failed: '+e2.code,'err')}}}
  if(ok){
    toast(`✅ Uploaded to ${used}!`);
    document.getElementById('lecTitle').value='';document.getElementById('lecSubject').value='';
    document.getElementById('lecLink').value='';document.getElementById('lecDate').value='';
    document.getElementById('lecImage').value='';
    document.getElementById('imgPreview').style.display='none';
    logA('lecture_upload',{title,subject:subj,db:used});
  }
  btn.disabled=false;btn.innerHTML='<i class="fas fa-upload"></i> Upload Lecture';
}

function delLec(id,source,title){
  openModal(`Delete "${title}"? This cannot be undone.`,async()=>{
    try{
      if(source==='db2')await db2.ref('lectures/'+id).remove();
      else await db1.ref('lectures/'+id).remove();
      toast('🗑️ Lecture deleted');
    }catch(e){toast('Delete failed: '+e.code,'err')}
  });
}

function cleanExpired(){

    if(!snap.exists()){toast('Nothing to clean','info');return}

    return dbU.ref().update(u);
  }).then(()=>toast('✅ Cleaned!')).catch(e=>toast('Error: '+e.code,'err'));
}
